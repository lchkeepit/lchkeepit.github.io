营业额统计

1588: [HNOI2002]营业额统计

Time Limit: 5 Sec Memory Limit: 162 MB
Submit: 5783 Solved: 1859

Description

营业额统计 Tiger最近被公司升任为营业部经理，他上任后接受公司交给的第一项任务便是统计并分析公司成立以来的营业情况。 Tiger拿出了公司的账本，账本上记录了公司成立以来每天的营业额。分析营业情况是一项相当复杂的工作。由于节假日，大减价或者是其他情况的时候，营业额会出现一定的波动，当然一定的波动是能够接受的，但是在某些时候营业额突变得很高或是很低，这就证明公司此时的经营状况出现了问题。经济管理学上定义了一种最小波动值来衡量这种情况： 该天的最小波动值 当最小波动值越大时，就说明营业情况越不稳定。 而分析整个公司的从成立到现在营业情况是否稳定，只需要把每一天的最小波动值加起来就可以了。你的任务就是编写一个程序帮助Tiger来计算这一个值。 第一天的最小波动值为第一天的营业额。  输入输出要求
Input

第一行为正整数 ，表示该公司从成立一直到现在的天数，接下来的n行每行有一个正整数 ，表示第i天公司的营业额。
Output

输出文件仅有一个正整数，即Sigma(每天最小的波动值) 。结果小于2^31 。
Sample Input

6
5
1
2
5
4
6
Sample Output

12
HINT

结果说明：5+|1-5|+|2-1|+|5-5|+|4-5|+|6-5|=5+4+1+0+1+1=12

题目分析

计算当前的波动值，需要找出过去的比当前值小的最大值，和比当前值大的最小值，最后选择两者和当前值的差最小的那一个。

代码

#include <cstdio>
#include <cstring>
#include <iostream>
#include <climits>
using namespace std;
#define fat(x) node[(x)].father
#define KEY(x) node[(x)].key
const int N = 100100;
struct Node
{
    int father;
    int son[2];
    int key;
}node[N];
int tot,root,n;
void newnode(int &r,int father,int key)
{
    r = ++tot;
    node[r].father = father;
    node[r].key = key;
    node[r].son[0] = node[r].son[1] = 0;
}
void rotate(int x,int kind)//0:left ,1:right
{
    int fa = fat(x);
    node[fa].son[1-kind] = node[x].son[kind];
    int st = node[x].son[kind];
    node[st].father = fa;
    if (node[fa].father)
        node[fat(fa)].son[node[fat(fa)].son[1] == fa]     = x;
    node[x].father = node[fa].father;
    node[x].son[kind] = fa;
    node[fa].father = x;
}
void splay(int r,int goal)//r to be goal.son
{
    while (node[r].father != goal)
    {
    if (fat((fat(r))) == goal) rotate(r,node[fat(r)].son[0] == r);
    else
    {
        int fa = fat(r);
        int kind = (node[fat(fa)].son[0] == fa);
        if (node[fa].son[kind] == r)
        {
            rotate(r,1-kind);
            rotate(r,kind);
        }
        else
        {
            rotate(fa,kind);
            rotate(r,kind);
        }
    }
}
if(goal == 0) root = r;
}
int insert(int k)
{
    int r = root;
    if (KEY(r) == k)
    {
        splay(r,0);
        return 0;
    }
    //printf("%d %d\n",r,node[r].son[0]);
    while (node[r].son[KEY(r) < k])
    {
        //puts("ww");
        if (KEY(r) == k)
        {
            splay(r,0);
            return 0;
        }
        r = node[r].son[KEY(r) < k];
    }
    newnode(node[r].son[KEY(r) < k],r,k);
    //printf("%d %d %d\n",node[r].son[0],r,KEY(r));
    splay(node[r].son[KEY(r) < k],0);
    return 1;
}
int get_pre(int x)
{
    int tmp = node[x].son[0];
    if (tmp == 0) return INT_MAX;
    while (node[tmp].son[1])
        tmp = node[tmp].son[1];
    return KEY(x)-KEY(tmp);
}
int get_next(int x)
{
    int tmp = node[x].son[1];
    if (tmp == 0) return INT_MAX;
    while (node[tmp].son[0])
        tmp = node[tmp].son[0];
    return KEY(tmp)-KEY(x);
}
int main()
{
    freopen("in.txt","r",stdin);
    while (~scanf("%d",&n))
    {
        tot = 0;
        root = 0;
        int ans = 0;
        for (int i = 1;i <= n;i++)
        {
            int num;
            if(scanf("%d",&num) == EOF) num = 0;
            if (i == 1)
            {
                ans += num;
                newnode(root,0,num);
                continue;
            }
            if (insert(num) == 0) continue;
            int a = get_pre(root);
            int b = get_next(root);
            ans += min(a,b);
        }
        printf("%d\n",ans);
    }
}
