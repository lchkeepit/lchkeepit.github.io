Play with Chain

Time Limit: 6000/2000 MS (Java/Others) Memory Limit: 65536/32768 K (Java/Others)
Total Submission(s): 2783 Accepted Submission(s): 1141

Problem Description

YaoYao is fond of playing his chains. He has a chain containing n diamonds on it. Diamonds are numbered from 1 to n.
At first, the diamonds on the chain is a sequence: 1, 2, 3, …, n.
He will perform two types of operations:
CUT a b c: He will first cut down the chain from the ath diamond to the bth diamond. And then insert it after the cth diamond on the remaining chain.
For example, if n=8, the chain is: 1 2 3 4 5 6 7 8; We perform “CUT 3 5 4”, Then we first cut down 3 4 5, and the remaining chain would be: 1 2 6 7 8. Then we insert “3 4 5” into the chain before 5th diamond, the chain turns out to be: 1 2 6 7 3 4 5 8.

FLIP a b: We first cut down the chain from the ath diamond to the bth diamond. Then reverse the chain and put them back to the original position.
For example, if we perform “FLIP 2 6” on the chain: 1 2 6 7 3 4 5 8. The chain will turn out to be: 1 4 3 7 6 2 5 8

He wants to know what the chain looks like after perform m operations. Could you help him?

Input

There will be multiple test cases in a test data.
For each test case, the first line contains two numbers: n and m (1≤n, m≤3*100000), indicating the total number of diamonds on the chain and the number of operations respectively.
Then m lines follow, each line contains one operation. The command is like this:
CUT a b c // Means a CUT operation, 1 ≤ a ≤ b ≤ n, 0≤ c ≤ n-(b-a+1).
FLIP a b // Means a FLIP operation, 1 ≤ a < b ≤ n.
The input ends up with two negative numbers, which should not be processed as a case.

Output

For each test case, you should print a line with n numbers. The ith number is the number of the ith diamond on the chain.

Sample Input

8 2 CUT 3 5 4 FLIP 2 6 -1 -1

Sample Output

1 4 3 7 6 2 5 8

Source
2010 ACM-ICPC Multi-University Training Contest（5）——Host by BJTU

题目分析

这道题需要完成的操作有两个，第一个是cut，这里便是类似于区间修改，比如我们要把[a,b]这个区间删除掉，那么我们要做的有三步

1 找到a-1，b+1对应的节点编号，这个用getrank可以做到

2 将a-1节点splay到根上，将b+1节点旋转到根的右儿子上，那么很容易发现我们需要上删除的区间变成了b+1节点的左儿子

3 直接把b+1的左儿子设为空便可以删除了，区间修改同理。为了节约空间，我们将删除的那些节点存入内存池。

对于翻转操作，其实和删除是一样的，最后一步将删除节点改为传递一个翻转的lazy标记即可

代码

#include <cstdio>
#include <iostream>

using namespace std;
#define lson ch[r][0]
#define rson ch[r][1]
#define keyvalue ch[ch[root][1]][0]
const int N = 300100;
int ch[N][2],rev[N],size[N],pre[N];
int key[N];
int root,n,m;
int tot;
inline void pushadd(int r)
{
    if (r == 0) return;
    swap(lson,rson);
    rev[r] ^= 1;
}
inline void pushup(int r)
{
    size[r] = size[lson] + size[rson] + 1;
}
void pushdown(int r)
{
    if (r == 0) return;
    if (rev[r])
    {
        pushadd(lson);
        pushadd(rson);
        rev[r] = 0;
    }
}    
void newnode(int &r,int father,int k)
{
    r = k;
    size[r] = 1;
    rev[r] = 0;
    key[r] = k;
    pre[r] = father;
    lson = rson = 0;
}
void build(int &r,int L,int R,int father)
{
    if (L>R) return;
    int m = (L + R) >> 1;
    newnode(r,father,m);
    build(lson,L,m-1,r);
    build(rson,m+1,R,r);
    pushup(r);
}
void init()
{
    key[root] = tot = root = ch[root][0] = ch[root][1]= size[root] = rev[root] = 0;
    newnode(root,0,n+1);
    newnode(ch[root][1],root,n+2);
    build(keyvalue,1,n,ch[root][1]);
    pushup(ch[root][1]);
    pushup(root);
}
void rotate(int r,int kind)
{
    int y = pre[r];
    pushdown(y);
    pushdown(r);
    ch[y][!kind] = ch[r][kind];
    pre[ch[r][kind]] = y;
    if (pre[y])
        ch[pre[y]][ch[pre[y]][1] == y] = r;
    ch[r][kind] = y;
    pre[r] = pre[y];
    pre[y] = r;
    pushup(y);
}
void splay(int r,int goal)
{
    pushdown(r);
    while (pre[r] != goal)
    {
        if (pre[pre[r]] == goal)
        {
            pushdown(pre[r]);
            pushdown(r);
            int kind = ch[pre[r]][0] == r;
        rotate(r,kind);
        }
        else
        {
            int y = pre[r];
            pushdown(pre[y]);
            pushdown(y);
            pushdown(r);
            int kind = ch[pre[y]][0] == y;
            if (ch[y][kind] == r)
            {
                rotate(r,!kind);
                rotate(r,kind);
            }
            else
            {
                rotate(y,kind);
                rotate(r,kind);
            }

        }
    }
pushup(r);
    if (goal == 0) root = r;
}

int getrank(int r,int x)
{
    pushdown(r);
    int t = size[lson];
    if (x == t + 1) return r;
    if (x <= t) return getrank(lson,x);
    return getrank(rson,x - t - 1);
}
void cut(int a,int b,int c)
{
    a = getrank(root,a);
    b = getrank(root,b+2);
    splay(a,0);
    splay(b,root);
    int r = ch[b][0];
    ch[b][0] = 0;
    pushup(b);
    pushup(root);
    int r1 = getrank(root,c+1);
    int r2 = getrank(root,c+2);
    splay(r1,0);
    splay(r2,root);
    ch[r2][0] = r;
    pre[r] = r2;
    pushup(r2);
    pushup(r1);
}    
void flip(int a,int b)
{    
    a = getrank(root,a);
    b = getrank(root,b+2);
    splay(a,0);
    splay(b,root);
    pushadd(keyvalue);
}    
int cnt = 0;
void d(int u)
{
    if (u == 0) return;
    pushdown(u);
    d(ch[u][0]);
    if (key[u] > 0 && key[u] <= n)
        printf("%d%c",key[u]," \n"[(++cnt) == n]);
    d(ch[u][1]);
}
char op[10];
int main()
{
    freopen("in.txt","r",stdin);
    while (~scanf("%d %d",&n,&m))
    {
        if (n == m && n == -1) break;
        init();
       // d(root);
       int a,b,c;
        for (int i = 1;i <= m;i++)
        {

            scanf("%s",op);
            if (op[0] == 'C')
            {

                scanf("%d %d %d",&a,&b,&c);
                cut(a,b,c);
            }
            else
            {
                scanf("%d %d",&a,&b);
                flip(a,b);
            }
        }

        //printf("%d\n",rev[keyvalue]);
        cnt = 0;
        d(root);
    }
}

营业额统计

1588: [HNOI2002]营业额统计

Time Limit: 5 Sec Memory Limit: 162 MB
Submit: 5783 Solved: 1859
[Submit][Status]
Description

营业额统计 Tiger最近被公司升任为营业部经理，他上任后接受公司交给的第一项任务便是统计并分析公司成立以来的营业情况。 Tiger拿出了公司的账本，账本上记录了公司成立以来每天的营业额。分析营业情况是一项相当复杂的工作。由于节假日，大减价或者是其他情况的时候，营业额会出现一定的波动，当然一定的波动是能够接受的，但是在某些时候营业额突变得很高或是很低，这就证明公司此时的经营状况出现了问题。经济管理学上定义了一种最小波动值来衡量这种情况： 该天的最小波动值 当最小波动值越大时，就说明营业情况越不稳定。 而分析整个公司的从成立到现在营业情况是否稳定，只需要把每一天的最小波动值加起来就可以了。你的任务就是编写一个程序帮助Tiger来计算这一个值。 第一天的最小波动值为第一天的营业额。  输入输出要求
Input

第一行为正整数 ，表示该公司从成立一直到现在的天数，接下来的n行每行有一个正整数 ，表示第i天公司的营业额。
Output

输出文件仅有一个正整数，即Sigma(每天最小的波动值) 。结果小于2^31 。
Sample Input

6
5
1
2
5
4
6
Sample Output

12
HINT

结果说明：5+|1-5|+|2-1|+|5-5|+|4-5|+|6-5|=5+4+1+0+1+1=12

题目分析

计算当前的波动值，需要找出过去的比当前值小的最大值，和比当前值大的最小值，最后选择两者和当前值的差最小的那一个。

代码

#include <cstdio>
#include <cstring>
#include <iostream>
#include <climits>
using namespace std;
#define fat(x) node[(x)].father
#define KEY(x) node[(x)].key
const int N = 100100;
struct Node
{
    int father;
    int son[2];
    int key;
}node[N];
int tot,root,n;
void newnode(int &r,int father,int key)
{
    r = ++tot;
    node[r].father = father;
    node[r].key = key;
    node[r].son[0] = node[r].son[1] = 0;
}
void rotate(int x,int kind)//0:left ,1:right
{
    int fa = fat(x);
    node[fa].son[1-kind] = node[x].son[kind];
    int st = node[x].son[kind];
    node[st].father = fa;
    if (node[fa].father)
        node[fat(fa)].son[node[fat(fa)].son[1] == fa]     = x;
    node[x].father = node[fa].father;
    node[x].son[kind] = fa;
    node[fa].father = x;
}
void splay(int r,int goal)//r to be goal.son
{
    while (node[r].father != goal)
    {
    if (fat((fat(r))) == goal) rotate(r,node[fat(r)].son[0] == r);
    else
    {
        int fa = fat(r);
        int kind = (node[fat(fa)].son[0] == fa);
        if (node[fa].son[kind] == r)
        {
            rotate(r,1-kind);
            rotate(r,kind);
        }
        else
        {
            rotate(fa,kind);
            rotate(r,kind);
        }
    }
}
if(goal == 0) root = r;
}
int insert(int k)
{
    int r = root;
    if (KEY(r) == k)
    {
        splay(r,0);
        return 0;
    }
    //printf("%d %d\n",r,node[r].son[0]);
    while (node[r].son[KEY(r) < k])
    {
        //puts("ww");
        if (KEY(r) == k)
        {
            splay(r,0);
            return 0;
        }
        r = node[r].son[KEY(r) < k];
    }
    newnode(node[r].son[KEY(r) < k],r,k);
    //printf("%d %d %d\n",node[r].son[0],r,KEY(r));
    splay(node[r].son[KEY(r) < k],0);
    return 1;
}
int get_pre(int x)
{
    int tmp = node[x].son[0];
    if (tmp == 0) return INT_MAX;
    while (node[tmp].son[1])
        tmp = node[tmp].son[1];
    return KEY(x)-KEY(tmp);
}
int get_next(int x)
{
    int tmp = node[x].son[1];
    if (tmp == 0) return INT_MAX;
    while (node[tmp].son[0])
        tmp = node[tmp].son[0];
    return KEY(tmp)-KEY(x);
}
int main()
{
    freopen("in.txt","r",stdin);
    while (~scanf("%d",&n))
    {
        tot = 0;
        root = 0;
        int ans = 0;
        for (int i = 1;i <= n;i++)
        {
            int num;
            if(scanf("%d",&num) == EOF) num = 0;
            if (i == 1)
            {
                ans += num;
                newnode(root,0,num);
                continue;
            }
            if (insert(num) == 0) continue;
            int a = get_pre(root);
            int b = get_next(root);
            ans += min(a,b);
        }
        printf("%d\n",ans);
    }
}

Splay

splay是一种独特的二叉排序树，通过rotate和splay操作，将需要处理的节点旋转至跟或是根的儿子处，在对其进行操作。也就是说splay天生就有一种类似缓存的能力。splay不能保证单次操作的复杂度O（logn）但是它能保证它的均摊复杂度o(logn).

rotate操作

void rotate(int r,int kind)
{
    int y = pre[r];
    pushdown(y);
    pushdown(r);
    ch[y][!kind] = ch[r][kind];
    pre[ch[r][kind]] = y;
    if (pre[y])
        ch[pre[y]][ch[pre[y]][1] == y] = r;
    ch[r][kind] = y;
    pre[r] = pre[y];
    pre[y] = r;
    pushup(y);
}

rotate分为左旋和右旋，这里用kind巧妙的减少了代码的冗余

Splay操作

void splay(int r,int goal)
{
    pushdown(r);
    while (pre[r] != goal)
    {
        if (pre[pre[r]] == goal)
        {
            pushdown(pre[r]);
            pushdown(r);
            int kind = ch[pre[r]][0] == r;
            rotate(r,kind);
        }
        else
        {
            int y = pre[r];
            pushdown(pre[y]);
            pushdown(y);
            pushdown(r);
            int kind = ch[pre[y]][0] == y;
            if (ch[y][kind] == r)
            {
                rotate(r,!kind);
                rotate(r,kind);
            }
            else
            {
                rotate(y,kind);
                rotate(r,kind);
            }

        }
    }
        pushup(r);
    if (goal == 0) root = r;
}

大家注意到这里pushdown的非常频繁，这是因为splay能实现一个lazy标记的翻转操作，翻转之后左右儿子需要交换才行

另外在翻转操作传递lazy标记的时候

void pushrev(int r)
{
    if (r == 0) return ;
    swap(ch[r][0],ch[r][1]);
    rev[r] ^= 1;
}

一定要先把当前节点翻转，才打标记，容易遗忘，类比线段树。
